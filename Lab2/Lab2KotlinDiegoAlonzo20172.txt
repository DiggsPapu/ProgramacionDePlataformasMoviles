fun processList(inputList: List<Any?>?): ArrayList<ItemData>? {
	
    var newList: ArrayList<ItemData> = arrayListOf(ItemData(0,89,"Entero", "-"))
    newList.removeAt(0)
    var counter = 0
    for (item: Any? in inputList.orEmpty()){
         
        //Dependiendo de que tipo de valor sea
        if (item is Int){
            if (item % 10 == 0){
                var value = ItemData(counter, item,"entero", "M10")
                newList.add(value)
            }
            else if (item % 5 == 0){
                var value = ItemData(counter, item,"entero", "M5")
                newList.add(value)
            }
            else if (item % 2 == 0){
                var value = ItemData(counter, item,"entero", "M2")
                newList.add(value)
            }
            else {
                var value = ItemData(counter, item,"entero", "-")
                newList.add(value)
            }
            
        }
        else if (item is String){
            var valCad = "L"+item.length
            var value = ItemData(counter, item, "cadena", valCad )
            newList.add(value)
            
        }
        else if (item == true){
            var value = ItemData(counter, item, "booleano", "verdadero" )
            newList.add(value)
        }
        else if (item == false){
            var value = ItemData(counter, item, "booleano", "falso" )
            newList.add(value)
        }
        else if (item == null){
            
        }
        else {
            var value = ItemData(counter, item, "desconocido", "desconocido" )
            newList.add(value)
        }
        //var value = ItemData(counter, item, )
        counter++
    }
    
    return newList
}
